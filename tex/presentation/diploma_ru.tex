\begin{frame}{Постановка задачи}
  \begin{columns}[T]

    % ── Левая: проблема ─────────────────────────────────────
    \begin{column}{0.50\textwidth}
      \textbf{Проблема:} классическая калибровка медленна\\[0.5em]
      \begin{enumerate}\small
        \item Вычислить характеристическую функцию $\phi(u;\boldsymbol\theta)$
        \item Оценить 88 опционов через FFT Карра--Мадана
        \item Подать MSE в оптимизатор Левенберга--Марквардта
        \item Повторить $\sim$10\,000 раз до сходимости
      \end{enumerate}
      \vspace{0.5em}
      \alert{$\Rightarrow$ \textbf{10–120 секунд} на одну калибровку}\\[0.3em]
      \alert{$\Rightarrow$ Нет аналитического якобиана}\\[0.5em]

      \begin{block}{Практические требования}
        \small
        \begin{itemize}
          \item[\faCheck] Реал-тайм трейдинг: жёсткий дедлайн $< 150$\,мс
          \item[\faCheck] Точные производные второго порядка (Греки)
          \item[\faCheck] Гарантия физических ограничений (усл.\ Феллера)
        \end{itemize}
      \end{block}
    \end{column}

    % ── Правая: идея решения ────────────────────────────────
    \begin{column}{0.46\textwidth}
      \begin{alertblock}{Цель дипломной работы}
        \scriptsize
        Создать \textbf{универсальный ML-фреймворк} для калибровки различных моделей
        ценообразования и оценить его применимость в продакшене.\\
        \vspace{0.3em}
        \textit{Суррогат Хестона — первый шаг реализации (Proof of Concept).}
      \end{alertblock}
      \vspace{0.4em}
      \textbf{Задачи реализации (на примере Хестона):}
      \begin{enumerate}\scriptsize
        \item[\textbf{Ф1}] Обучить MLP-суррогат
                           $\mathcal{N}_w : \mathbb{R}^5 \to \mathbb{R}^{88}$
                           в пространстве суммарной дисперсии $W$
        \item[\textbf{Ф2}] Реализовать байесовскую оценку неопределённости
                           (MC Dropout)
        \item[\textbf{Ф3}] Обучить LSTM для прогнозирования
                           динамики параметров на следующий день
      \end{enumerate}

    % ── Правая: ключевые решения ────────────────────────────
    \begin{column}{0.46\textwidth}
      \begin{block}{Пространство суммарной дисперсии $W$}
        \small
        Сеть предсказывает не IV напрямую, а:
        \[ W(K,T) = \sigma_{\mathrm{IV}}^2(K,T)\cdot T \]
        \begin{itemize}
          \item[\faCheck] Линеаризует уравнение Дюпира
          \item[\faCheck] Условие без арбитража:
                $\partial W/\partial T \ge 0$\\
                (критерий Карра--Мадана)
          \item[\faCheck] Стабилизирует градиенты при обучении
        \end{itemize}
      \end{block}
      \vspace{0.25em}
      \begin{alertblock}{Зачем ELU, а не ReLU?}
        \scriptsize
        ReLU: $f''(x) \equiv 0 \Rightarrow$ гессиан нулевой, греки \textbf{бессмысленны}\\[0.2em]
        ELU: $f''(x) \neq 0 \Rightarrow$ корректный гессиан\\[-0.1em]
        $\|H_\mathrm{ELU}\|_F = 1.039 \gg \|H_\mathrm{ReLU}\|_F = 0$
      \end{alertblock}
    \end{column}

  \end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════════
\begin{frame}{Фаза 2: байесовская оценка неопределённости (MC Dropout)}
  \begin{columns}[T]

    % ── Левая: метод ────────────────────────────────────────
    \begin{column}{0.50\textwidth}
      \textbf{Зачем оценивать неопределённость?}\\[0.3em]
      {\small Точечная калибровка не говорит, насколько
      уверена модель в своём предсказании.}\\[0.5em]

      \begin{block}{Алгоритм MC Dropout \small(Gal \& Ghahramani, 2016)}
        \small
        \begin{enumerate}
          \item Оставить \texttt{Dropout(p=0.1)} \textbf{активным} на инференсе
                (\texttt{model.train()})
          \item Выполнить $N = 100$ стохастических forward-pass
          \item Вычислить среднее $\mu$ и стд $\sigma$ по семплам
        \end{enumerate}
        \[\hat{\boldsymbol\sigma} \pm 2\sigma_{\mathrm{MC}}\]
        $\Rightarrow$ \textbf{доверительный интервал} поверхности IV
      \end{block}

      \vspace{0.2em}
      \begin{block}{Математическая интерпретация}
        \small
        Байесовское приближение над весами:
        $p(\boldsymbol\sigma^*\mid\boldsymbol\theta,\mathcal{D})
          \approx \frac{1}{N}\sum_{i=1}^{N}\mathcal{N}_{w_i}(\boldsymbol\theta)$
      \end{block}
    \end{column}

    % ── Правая: результаты ──────────────────────────────────
    \begin{column}{0.46\textwidth}
      \textbf{Схема 100 forward-pass:}\\[0.4em]
      \begin{center}
      \begin{tikzpicture}[
        box/.style={rectangle, rounded corners=3pt, align=center,
                    font=\scriptsize, minimum height=0.65cm},
        arr/.style={->, thick},
      ]
        \node[box, draw=hestonblue, fill=hestonblue!10,
              minimum width=3.2cm] (net) at (0,0)
          {MLP с активным\\Dropout};
        \node[box, draw=gray!50, fill=gray!8,
              minimum width=1.3cm] (s1) at (-1.8,-1.4) {$\hat{W}^{(1)}$};
        \node[box, draw=gray!50, fill=gray!8,
              minimum width=1.3cm] (s2) at (0,   -1.4) {$\hat{W}^{(i)}$};
        \node[box, draw=gray!50, fill=gray!8,
              minimum width=1.3cm] (sN) at ( 1.8,-1.4) {$\hat{W}^{(N)}$};
        \node[font=\tiny, color=gray] at (0.9,-1.4) {$\cdots$};
        \node[box, draw=hestonorange, fill=hestonorange!10,
              minimum width=3.2cm] (out) at (0,-2.7)
          {$\mu \pm 2\sigma_{\mathrm{MC}}$\\IV-поверхность с полосами};
        \draw[arr, hestonblue] (net) -- (s1);
        \draw[arr, hestonblue] (net) -- (s2);
        \draw[arr, hestonblue] (net) -- (sN);
        \draw[arr, hestonorange] (s1) -- (out);
        \draw[arr, hestonorange] (s2) -- (out);
        \draw[arr, hestonorange] (sN) -- (out);
        \node[font=\tiny, color=hestonblue, left=0.05cm of net]
          {\rotatebox{90}{$N\!=\!100$ проходов}};
      \end{tikzpicture}
      \end{center}

      \vspace{0.3em}
      \begin{exampleblock}{Результаты ($N=100$)}
        \scriptsize
        \begin{tabular}{@{}ll@{}}
          Макс.\ $\sigma_{\mathrm{MC}}$ (ATM, $T=1$) & $< 5\times10^{-4}$ \\
          Ширина полос $\pm 2\sigma$                  & $< 0.1\%$ IV \\
          Накладные расходы                           & $\approx100\times$ \\
        \end{tabular}\\[0.3em]
        \textit{Узкие полосы подтверждают отсутствие переобучения.}
      \end{exampleblock}
    \end{column}

  \end{columns}
\end{frame}

% Слайд 9 — Фаза 3: LSTM-динамика параметров
% ══════════════════════════════════════════════════════════════
\begin{frame}{Фаза 3: LSTM-прогноз динамики параметров Хестона}
  \begin{columns}[T]

    % ── Левая: архитектура + датасет ────────────────────────
    \begin{column}{0.50\textwidth}
      \textbf{Задача:} предсказать $\hat{\boldsymbol\theta}_{t+1}$ по истории
      поверхностей $W$ за 10 дней.\\[0.4em]

      \textbf{Архитектура:}
      \[
        \underbrace{(10 \times 88)}_{\text{10-дневная история }W}
        \xrightarrow{\;\text{LSTM}_{64}^{2}\;}
        \underbrace{\hat{\boldsymbol\theta}_{t+1}}_{5\text{ параметров}}
      \]
      \begin{center}\scriptsize
      \begin{tabular}{@{}ll@{}}
        \toprule
        Параметр          & Значение \\ \midrule
        LSTM слоёв        & 2, hidden = 64 \\
        Инициализация     & Xavier (вход), Ортогональная (рекурр.) \\
        Forget-gate bias  & = 1 (стандартный трюк) \\
        Нормализация      & LayerNorm после LSTM \\
        Обучаемых пар-ов  & 73\,157 \\
        Оптимизатор       & AdamW + CosineAnnealingLR \\
        Ранняя остановка  & EarlyStopping (patience=20) \\
        \bottomrule
      \end{tabular}
      \end{center}
    \end{column}

    % ── Правая: данные и pipeline ───────────────────────────
    \begin{column}{0.46\textwidth}
      \begin{block}{Генерация обучающих данных}
        \small
        \begin{enumerate}
          \item \textbf{600 траекторий} параметров по процессам\\
                Орнштейна--Уленбека (калибр.\ к SPX)
          \item Каждый день: $\boldsymbol\theta_t
                \xrightarrow{\text{MLP}} W_t$ (88 точек)
          \item \textbf{102\,000 окон}: $(W_{t-9},\ldots,W_t)
                \to \boldsymbol\theta_{t+1}$
          \item Разбивка 80/10/10 по границам траекторий\\
                (\textit{утечки данных нет})
        \end{enumerate}
      \end{block}

      \vspace{0.4em}
      \begin{alertblock}{Z-нормализация меток}
        \small
        Метки $\boldsymbol\theta$ нормализуются по статистике\\
        обучающей выборки (Z-score).\\[0.2em]
        На инференсе: обратное преобразование +\\
        жёсткая привязка к физическим границам\\
        (условие Феллера гарантировано).
      \end{alertblock}
    \end{column}

  \end{columns}
\end{frame}
